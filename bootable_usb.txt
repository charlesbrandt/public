Might be nice to reformat the existing drive to give it a label of which version is on the device:

#CHANGE DEV LOCATION!
#check where your usb is mounted:
dmesg
mount

sudo ls #prime sudo

#this is different than disks, but important!
sudo parted -a optimal /dev/sdf mklabel msdos

sudo parted -a optimal /dev/sdf rm 1
sudo parted -a optimal /dev/sdf rm 2

#1MB is the start, 2GB is the end
#sudo parted -a optimal /dev/sdf mkpart primary fat32 1MB 2GB
sudo parted -a optimal /dev/sdf mkpart primary fat32 1MB 100%
sudo mkfs.vfat -n SHARE /dev/sdf1

sudo parted -a optimal /dev/sdf set 1 boot on

#not sure if it's necessary to turn off other flags:
sudo parted -a optimal /dev/sdf set 1 esp off

sudo parted -a optimal /dev/sdf print
