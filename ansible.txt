#first, set up the share on VirtualBox via:

Devices -> Shared Folders -> Shared Folder Settings...

Click the folder with the green plus to the right.


Using this requires the VirtualBox Guest Additions addons to be installed on the client

Devices > Insert Guest Additions CD image...

sudo mount /dev/cdrom /media/cdrom
sudo /media/cdrom/VBoxLinuxAdditions.run

lineinfile: dest=/etc/modules line="vboxsf"


sudo vi /etc/fstab
#and follow pattern for /c
c /c vboxsf uid=1000,gid=1000,auto 0 0
DATA /media/charles/DATA vboxsf uid=1000,gid=1000,auto 0 0
